    ```javascript
    #!/usr/bin/env node
    /**
     * jscodeshift codemod: replace inline `fontSize: N` literals in JSX style props
     * with `...typeScale.<token>` spread.
     *
     * Usage:  pnpm codemod:type-scale          (dry-run, prints diff)
     *         pnpm codemod:type-scale:apply    (writes changes)
     *
     * Mapping (locked by UI-SPEC §P1 typeScale):
     *   11 → typeScale.micro
     *   12 → typeScale.caption
     *   14 → typeScale.label
     *   17 → typeScale.body
     *   18 → typeScale.propertiesTitle  (added in Phase 15)
     *   20 → typeScale.sectionTitle
     *   32 → typeScale.pageTitle
     *   36 → typeScale.documentTitle    (added in Phase 15)
     *
     * SKIP: files containing the comment marker `// codemod-skip-fontsize` (per-file opt-out).
     */

    const FONT_SIZE_MAP = {
      11: 'micro',
      12: 'caption',
      14: 'label',
      17: 'body',
      18: 'propertiesTitle',
      20: 'sectionTitle',
      32: 'pageTitle',
      36: 'documentTitle',
    };

    export default function transformer(file, api) {
      const j = api.jscodeshift;
      const root = j(file.source);

      if (file.source.includes('codemod-skip-fontsize')) {
        return null;
      }

      // Find all ObjectProperty nodes with key.name === 'fontSize' and a numeric Literal value
      const props = root.find(j.ObjectProperty, {
        key: { name: 'fontSize' },
        value: { type: 'Literal' },
      });

      let modified = false;
      props.forEach((path) => {
        const valueNode = path.value.value;
        if (typeof valueNode.value !== 'number') return;
        const tokenName = FONT_SIZE_MAP[valueNode.value];
        if (!tokenName) return;

        // Replace `fontSize: N` with `...typeScale.<tokenName>`
        j(path).replaceWith(
          j.spreadElement(
            j.memberExpression(j.identifier('typeScale'), j.identifier(tokenName))
          )
        );
        modified = true;
      });

      return modified ? root.toSource({ quote: 'single' }) : null;
    }
    ```

    ```javascript
      48: { type: 'literal-member', object: 'spacing', property: '3xl' },
    ```

    ```powershell
      $hits = Select-String -Path $IncludeGlobs -Pattern $Pattern -ErrorAction SilentlyContinue
    ```
